<div class="container mt-5">
    <form method="post">
        <div class="form-group">
            <label asp-for="UserInput.InputText"></label>
            <input id="inputText" asp-for="UserInput.InputText" class="form-control" />
        </div>
        <button type="submit" class="btn btn-primary">Copy to Clipboard</button>
    </form>
    <div class="mt-3">
        <div id="copySuccessMessage" class="alert alert-success alert-dismissible fade show d-none" role="alert">
            Copied to Clipboard!
            <button id="closeButton" type="button" class="btn-close" data-bs-dismiss="alert" aria-label="Close"></button>
        </div>
    </div>
</div>

<script>
    document.getElementById('inputText').addEventListener('input', function () {
        // Handle input changes, if necessary
    });

    document.querySelector('form').addEventListener('submit', function (e) {
        e.preventDefault();
        var inputText = document.getElementById('inputText').value;
        var tempInput = document.createElement('input');
        tempInput.value = inputText;
        document.body.appendChild(tempInput);
        tempInput.select();
        document.execCommand('copy');
        document.body.removeChild(tempInput);

        var copySuccessMessage = document.getElementById('copySuccessMessage');
        copySuccessMessage.classList.remove('d-none', 'fade');
        setTimeout(function() {
            copySuccessMessage.classList.add('fade');
            setTimeout(function() {
                copySuccessMessage.classList.add('d-none');
            }, 1000);
        }, 3000);
    });

    document.getElementById('closeButton').addEventListener('click', function (event) {
        event.preventDefault(); // Prevent default close behavior
        var copySuccessMessage = document.getElementById('copySuccessMessage');
        copySuccessMessage.classList.add('d-none', 'fade');
    });
</script>
